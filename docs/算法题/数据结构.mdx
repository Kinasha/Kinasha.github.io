- 树和数组互相转换

```ts
type Node = {
  id: number;
  children: Node[];
  pid: number;
};

function isRootNode(node: Node) {
  return node.pid === 0;
}

function arr2Tree(arr: Node[]) {
  const tree: Node[] = [];
  for (let i = 0; i < arr.length; i++) {
    if (isRootNode(arr[i])) {
      tree.push(arr[i]);
    } else {
      const parent = arr.find((item) => item.id === arr[i].pid);
      parent?.children.push(arr[i]);
    }
  }
  return tree;
}
function tree2Arr(tree: Node[]) {
  const arr: Node[] = [];
  const dfsHelper = (root: Node) => {
    if (root.children.length === 0) {
      arr.push({ id: root.id, children: [], pid: root.pid });
    } else {
      for (const child of root.children) {
        dfsHelper(child);
      }
      arr.push({ id: root.id, children: [], pid: root.pid });
    }
  };
  for (const root of tree) {
    dfsHelper(root);
  }
  return arr;
}
```
